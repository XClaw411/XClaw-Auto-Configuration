 一、基础配置（~/.openclaw/openclaw.json）                                                                                                                                                                                                 
                                                                                                                                                                                                                                           
 ### 1. 启用飞书插件                                                                                                                                                                                                                       
                                                                                                                                                                                                                                           
 ```json                                                                                                                                                                                                                                   
   {                                                                                                                                                                                                                                       
     "plugins": {                                                                                                                                                                                                                          
       "entries": {                                                                                                                                                                                                                        
         "feishu": {                                                                                                                                                                                                                       
           "enabled": false      // 旧插件，禁用                                                                                                                                                                                           
         },                                                                                                                                                                                                                                
         "openclaw-lark": {                                                                                                                                                                                                                
           "enabled": true       // ✅ 启用新插件                                                                                                                                                                                          
         }                                                                                                                                                                                                                                 
       },                                                                                                                                                                                                                                  
       "allow": ["openclaw-lark"]                                                                                                                                                                                                          
     }                                                                                                                                                                                                                                     
   }                                                                                                                                                                                                                                       
 ```                                                                                                                                                                                                                                       
                                                                                                                                                                                                                                           
 ### 2. 配置飞书渠道（关键部分）                                                                                                                                                                                                           
                                                                                                                                                                                                                                           
 ```json                                                                                                                                                                                                                                   
   {                                                                                                                                                                                                                                       
     "channels": {                                                                                                                                                                                                                         
       "feishu": {                                                                                                                                                                                                                         
         "enabled": true,                                                                                                                                                                                                                  
         "appId": "cli_a901acaf1df8dbd9",                                                                                                                                                                                                  
                                                                                                                                                                                                                                           
         // ✅ 解决方案：直接内嵌 appSecret，避免外部文件读取导致断开                                                                                                                                                                      
         "appSecret": "w5rNsogod30AfcwV5n6e9bayJsVGHTaA",                                                                                                                                                                                  
                                                                                                                                                                                                                                           
         "domain": "feishu",                                                                                                                                                                                                               
         "connectionMode": "websocket",                                                                                                                                                                                                    
                                                                                                                                                                                                                                           
         // ✅ 不需要@提及                                                                                                                                                                                                                 
         "requireMention": false,                                                                                                                                                                                                          
                                                                                                                                                                                                                                           
         // ✅ 私聊开放给所有人                                                                                                                                                                                                            
         "dmPolicy": "open",                                                                                                                                                                                                               
         "allowFrom": ["*"],                                                                                                                                                                                                               
                                                                                                                                                                                                                                           
         // ✅ 群聊开放                                                                                                                                                                                                                    
         "groupPolicy": "open"                                                                                                                                                                                                             
       }                                                                                                                                                                                                                                   
     }                                                                                                                                                                                                                                     
   }                                                                                                                                                                                                                                       
 ```                                                                                                                                                                                                                                       
                                                                                                                                                                                                                                           
 │ ⚠️ 重要：appSecret 必须直接写在配置里，不能用 {"source": "file", ...} 方式，否则每次读取会被误判为"凭证变更"导致 WebSocket 断开！

 二、解决"发一条就断"问题                                                                                                                                                                                                                  
                                                                                                                                                                                                                                           
 ### 问题现象                                                                                                                                                                                                                              
                                                                                                                                                                                                                                           
 ```                                                                                                                                                                                                                                       
   credentials changed, disposing stale instance                                                                                                                                                                                           
   disconnecting WebSocket                                                                                                                                                                                                                 
   ws client closed manually (force)                                                                                                                                                                                                       
 ```                                                                                                                                                                                                                                       
                                                                                                                                                                                                                                           
 ### 根本原因                                                                                                                                                                                                                              
                                                                                                                                                                                                                                           
 使用外部文件存储凭证时，系统每次收到消息都重新读取文件，检测到" 变化"后强制断开 WebSocket 重连。                                                                                                                                          
                                                                                                                                                                                                                                           
 ### 解决方案                                                                                                                                                                                                                              
                                                                                                                                                                                                                                           
 将凭证直接内嵌到配置文件中：                                                                                                                                                                                                              
                                                                                                                                                                                                                                           
 ❌ 错误方式（导致断开）：                                                                                                                                                                                                                 
                                                                                                                                                                                                                                           
 ```json                                                                                                                                                                                                                                   
   "appSecret": {                                                                                                                                                                                                                          
     "source": "file",                                                                                                                                                                                                                     
     "provider": "lark-secrets",                                                                                                                                                                                                           
     "id": "/lark/appSecret"                                                                                                                                                                                                               
   }                                                                                                                                                                                                                                       
 ```                                                                                                                                                                                                                                       
                                                                                                                                                                                                                                           
 ✅ 正确方式（稳定连接）：                                                                                                                                                                                                                 
                                                                                                                                                                                                                                           
 ```json                                                                                                                                                                                                                                   
   "appSecret": "w5rNsogod30AfcwV5n6e9bayJsVGHTaA"                                                                                                                                                                                         
 ```

 三、配置检查与修复                                                                                                                                                                                                                        
                                                                                                                                                                                                                                           
 ### 运行 Doctor 检查                                                                                                                                                                                                                      
                                                                                                                                                                                                                                           
 ```bash                                                                                                                                                                                                                                   
   openclaw doctor                                                                                                                                                                                                                         
 ```                                                                                                                                                                                                                                       
                                                                                                                                                                                                                                           
 ### 自动修复命令                                                                                                                                                                                                                          
                                                                                                                                                                                                                                           
 ```bash                                                                                                                                                                                                                                   
   openclaw doctor --fix                                                                                                                                                                                                                   
 ```                                                                                                                                                                                                                                       
                                                                                                                                                                                                                                           
 这会修复：                                                                                                                                                                                                                                
 - allowFrom: ["*"] （配合 dmPolicy="open"）
